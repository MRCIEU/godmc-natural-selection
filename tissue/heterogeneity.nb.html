Sometimes there are multiple SNPs for a CpG. Count how many heterogeneity Q stats are significant (FDR < 0.05) per CpG

outcpg <- out %>% 
  mutate(fdr=p.adjust(Qpval, "fdr")) %>%
  filter(!is.na(fdr)) %>%
  group_by(cpg) %>%
  summarise(
    n=n(), 
    nsig=sum(fdr < 0.05),
    min=min(fdr),
    max=max(fdr)
  )
